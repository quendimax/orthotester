косць -> os, ossis n

скрываўлены -> cruentus

паварочваць -> converto, converti, conversum, convertere  # = verto
дзівіцца -> miror, miratus sum, mirari
лажыцца -> recumbo, recubui, —, recumbere   # <-> surgo
стукаць -> percutio, percussi, percussum, percutere   # = pulso, pulsare
ламаць -> frango, fregi, fractum, frangere  # = rumpere (rem duram)
плакаць -> fleo, flevi, fletum, flere  # = lacrimo
цярпець -> patior, passus sum, pati
не ведаць, не знать -> ignoro, ignoravi, ignoratum, ignorare
даведвацца, пазнаваць -> nosco, novi, notum, noscere
жадаць -> cupio, cupivi, cupitum, cupere
